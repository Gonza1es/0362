#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main()
{
    setlocale(LC_ALL, "rus");//поддержка русского алфавита

    int n, m, k;
    printf("Введите кол-во элементов:");
    scanf ("%d", &n);
    int a[n];
    printf("Введите элементы:");
    for (int i=0; i<n; i++)
    {
        scanf("%d",&m);
        a[i]=m;
    }
    k=pow(2, n);//количество элементов множества
    for (int i=0; i<k; i++)
    {
        printf("{");
        for (int j=0; j<n; j++)
            if (i & (1<<j)) //выводится jый элемент, если существует jый бит
                printf("%d",a[j]);
                printf("}");
        printf("\n");
    }
    return 0;
}
