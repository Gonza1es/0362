#include <stdio.h>
#include <stdlib.h>

int main()
{
    int num, k;
    scanf("%d", &num);
    int n;
    n=1;
    k=0;
    while (!(num<n))
    {
        k=k+1;
        n=n*(n+1);
    }
    int d=1;
    for (int i=2; i<k+1; i++)
        d=i*d;
    int x, oc, m;
    m=k;
    for (int i=0; i<k; i++)
    {
        oc=num%d;
        x=(num-oc)/d;
        num=oc;
        d=d/m;
        m--;
        printf("%d", x);
    }
    return 0;
}
