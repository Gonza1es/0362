#include <stdio.h>
#include <stdlib.h>

int main()
{
    int num;
    scanf("%d", &num);
    int n;
    n=1;
    int sum=0;
    int p;
    while (num>0)
    {
        p=num%10;
        sum=p*n+sum;
        n=n*(n+1);
        num=num/10;
    }
    printf("%d", sum);
    return 0;
}
